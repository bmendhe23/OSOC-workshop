git init: "Create an empty Git repository or reinitialize an existing one" "USAGE: git init"
git add: "Add file contents to the staging index" "USAGE: git add __FILENAME__"
git commit: "Record changes to the repository" "USAGE: git commit -m __MESSAGE__"
git reset: "Reset current HEAD to the specified state" "USAGE: git reset"
git revert: "Revert some existing commits" "USAGE: git revert HEAD"
git status: "Show the working tree status" "USAGE: git status"
git diff: "Show changes between commits, commit and working tree, etc" "USAGE: git diff"
git log: "Show commit logs" "USAGE: git log --oneline"
git checkout: "Switch branches or restore working tree files" "USAGE: git checkout -b __BRANCH__ or git checkout __BRANCH__"
git merge: "Join two or more development histories together" "USAGE: git merge __BRANCH__"
